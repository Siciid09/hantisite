    <script>
        // Application State
        let currentCategory = 'invoice';
        let currentReportType = ''; // NEW: To store the selected report type
        let currentReportName = ''; // NEW: To store the display name

        document.addEventListener('DOMContentLoaded', () => {
            const categoryTabs = document.getElementById('category-tabs');

            // Panel Selectors
            const templatePanels = document.getElementById('template-panels');
            const invoicePanel = document.getElementById('invoice-panel');
            const generalPanel = document.getElementById('general-panel');
            
            // General Panel Sub-View Selectors
            const generalReportSelection = document.getElementById('general-report-selection');
            const generalTierSelection = document.getElementById('general-tier-selection');
            const backToReportsBtn = document.getElementById('back-to-reports-btn');
            
            // Tier Selection Elements
            const tierSelectionTitle = document.getElementById('tier-selection-title');
            const tierDefaultTitle = document.getElementById('tier-default-title');
            const tierModernTitle = document.getElementById('tier-modern-title');
            const tierPremiumTitle = document.getElementById('tier-premium-title');
            const tierDefaultBtn = document.getElementById('tier-default-btn');
            const tierModernBtn = document.getElementById('tier-modern-btn');
            const tierPremiumBtn = document.getElementById('tier-premium-btn');


            // --- Event Listener for Category Tabs ---
            categoryTabs.addEventListener('click', (e) => {
                const clickedTab = e.target.closest('button');
                if (!clickedTab) return; // Guard clause

                const newCategory = clickedTab.dataset.category;
                if (newCategory === currentCategory) return; // Already on this tab

                currentCategory = newCategory;

                // Update tab button styles
                Array.from(categoryTabs.children).forEach(tab => {
                    if (tab.dataset.category === currentCategory) {
                        tab.classList.add('active');
                        tab.classList.remove('inactive');
                    } else {
                        tab.classList.add('inactive');
                        tab.classList.remove('active');
                    }
                });
                
                // NEW: Show/Hide Panels
                if (currentCategory === 'invoice') {
                    invoicePanel.classList.remove('hidden');
                    generalPanel.classList.add('hidden');
                } else {
                    invoicePanel.classList.add('hidden');
                    generalPanel.classList.remove('hidden');
                }
                
                // NEW: Reset general panel to report selection
                generalTierSelection.classList.add('hidden');
                generalReportSelection.classList.remove('hidden');
            });


            // --- NEW: Event Listener for Report Type Selection ---
            generalReportSelection.addEventListener('click', (e) => {
                const clickedButton = e.target.closest('.report-select-btn');
                if (!clickedButton) return;
                
                currentReportType = clickedButton.dataset.reportType;
                currentReportName = clickedButton.dataset.reportName;

                // Update tier selection titles and button data
                tierSelectionTitle.innerText = `Select Tier for: ${currentReportName}`;
                tierDefaultTitle.innerText = `Default ${currentReportName}`;
                tierModernTitle.innerText = `Modern ${currentReportName}`;
                tierPremiumTitle.innerText = `Premium ${currentReportName}`;
                
                tierDefaultBtn.dataset.template = `${currentReportType}-default-template`;
                tierModernBtn.dataset.template = `${currentReportType}-modern-template`;
                tierPremiumBtn.dataset.template = `${currentReportType}-premium-template`;

                // Switch views
                generalReportSelection.classList.add('hidden');
                generalTierSelection.classList.remove('hidden');
            });
            
            // --- NEW: Event Listener for Back Button ---
            backToReportsBtn.addEventListener('click', () => {
                generalTierSelection.classList.add('hidden');
                generalReportSelection.classList.remove('hidden');
                currentReportType = '';
                currentReportName = '';
            });


            // --- MODIFIED: Event Listener for all Generate Buttons (using delegation) ---
            templatePanels.addEventListener('click', (e) => {
                const clickedButton = e.target.closest('.generate-btn');
                if (!clickedButton) return;
                
                const templateId = clickedButton.dataset.template;
                if (!templateId) return;

                generatePdf(templateId, clickedButton);
            });


            // --- NEW: PDF Generation Function ---
            function generatePdf(templateId, clickedButton) {
                // Show loader on the specific button
                const originalText = clickedButton.innerHTML;
                clickedButton.innerHTML = 'Generating...';
                clickedButton.disabled = true;

                // Find the correct template
                const element = document.getElementById(templateId);
                
                if (!element) {
                    console.error('Template not found:', templateId);
                    clickedButton.innerHTML = 'Error';
                    setTimeout(() => { // Reset after a delay
                        clickedButton.innerHTML = originalText;
                        clickedButton.disabled = false;
                    }, 2000);
                    return;
                }
                
                // Set PDF options
                const filename = `${templateId.replace('-template', '')}_${Date.now()}.pdf`;
                const opt = {
                    margin:       0,
                    filename:     filename,
                    image:        { type: 'jpeg', quality: 0.98 },
                    html2canvas:  { scale: 2, useCORS: true, letterRendering: true },
                    jsPDF:        { unit: 'in', format: 'letter', orientation: 'portrait' }
                };

                // Run PDF generation
                html2pdf().from(element).set(opt).save()
                    .then(() => {
                        console.log('PDF Generated successfully');
                        resetButton(clickedButton, originalText);
                    })
                    .catch((err) => {
                        console.error('PDF Generation Error:', err);
                        clickedButton.innerHTML = 'Error';
                        setTimeout(() => { // Reset after a delay
                            resetButton(clickedButton, originalText);
                        }, 2000);
                    });
            }
            
            // --- NEW: Reset Button Function ---
            function resetButton(button, originalText) {
                button.innerHTML = originalText;
                button.disabled = false;
            }
            
        }); // Closes DOMContentLoaded
    </script>
